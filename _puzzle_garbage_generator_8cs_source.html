92 private void AddGarbage()
93 {
94 var positions = FindOpenPositions();
95 var targetBlocks = Math.Min(TargetGarbageBlocks(positions.Count), positions.Count);
96 RandomSeed.Shuffle(positions);
97
98 for (int i = 0; i < targetBlocks; i++)
99 {
100 var position = positions[i];
101 Layout.Tiles[position] |= GetRandomBlock();
102 }
103 }

108 private Array2D<PuzzleTile> GetMarkedTiles()
109 {
110 var tiles = new Array2D<PuzzleTile>(Layout.Tiles);
111
112 foreach (var move in Layout.Moves)
113 {
114 tiles[move.PushTilePosition()] |= PuzzleTile.Marked;
115 tiles[move.StopTilePosition()] |= PuzzleTile.Marked;
116
117 var min = Vector2DInt.Min(move.FromPosition, move.ToPosition);
118 var max = Vector2DInt.Max(move.FromPosition, move.ToPosition);
119
120 for (int i = min.X; i <= max.X; i++)
121 {
122 for (int j = min.Y; j <= max.Y; j++)
123 {
124 tiles[i, j] |= PuzzleTile.Marked;
125 }
126 }
127 }
128
129 return tiles;
130 }
