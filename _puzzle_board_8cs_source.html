32 public static RaycastHit Raycast(Array2D<PuzzleTile> tiles, Vector2DInt position, Vector2DInt offset, PuzzleTile blockingLayers)
33 {
34 if (offset == Vector2DInt.Zero)
35 throw new ArgumentException("Offset must be non-zero vector.");
36
37 var tile = PuzzleTile.None;
38 blockingLayers |= PuzzleTile.OutOfBounds;
39
40 while ((tile & blockingLayers) == PuzzleTile.None)
41 {
42 position += offset;
43 tile = tiles.GetOrDefault(position.X, position.Y, PuzzleTile.OutOfBounds);
44 }
45
46 return new RaycastHit(tile, position);
47 }

67 public static string TilesToTiledString(IList<Array2D<PuzzleTile>> tiles, int columns, Func<PuzzleTile, char> tileToCharacter = null)
68 {
69 if (tiles.Count == 0)
70 return string.Empty;
71
72 tileToCharacter = tileToCharacter ?? TileToCharacter;
73 const int spaceCount = 3;
74 columns = Math.Max(columns, 1);
75 var builder = new StringBuilder();
76 var width = 2 * tiles[0].Columns + spaceCount;
77 var rows = (int)Math.Ceiling(tiles.Count / (double)columns);
78
79 for (int row = 0; row < rows; row++)
80 {
81 // Add headers
82 for (int column = 0; column < columns; column++)
83 {
84 var index = row * columns + column;
85
86 if (index >= tiles.Count)
87 break;
88
89 var header = index == 0 ? "Start board:" : $"Move {index}:";
90 builder.Append(header);
91
92 if (column < columns - 1)
93 {
94 var length = Math.Max(width - header.Length, 0);
95 builder.Append(' ', length);
96 }
97 }
98
99 builder.Append('\n');
100
101 // Add tile strings
102 for (int i = 0; i < tiles[0].Rows; i++)
103 {
104 for (int column = 0; column < columns; column++)
105 {
106 var index = row * columns + column;
107
108 if (index >= tiles.Count)
109 break;
110
111 // Add row tile strings
112 for (int j = 0; j < tiles[0].Columns; j++)
113 {
114 builder.Append(tileToCharacter.Invoke(tiles[index][i, j])).Append(' ');
115 }
116
117 // Add separator
118 if (column < columns - 1)
119 builder.Append(' ', spaceCount);
120 }
121
122 builder.Append('\n');
123 }
124
125 if (row < rows - 1)
126 builder.Append('\n');
127 }
128
129 return builder.ToString();
130 }

159 public static Array2D<PuzzleTile> StringsToTiles(IList<string> lines, Func<char, PuzzleTile> characterToTile = null)
160 {
161 if (lines.Count == 0 || lines[0].Length == 0)
162 return new Array2D<PuzzleTile>();
163
164 characterToTile = characterToTile ?? CharacterToTile;
165 var result = new Array2D<PuzzleTile>(lines.Count, lines[0].Length);
166
167 for (int i = 0; i < lines.Count; i++)
168 {
169 var line = lines[i];
170
171 if (line.Length != lines[0].Length)
172 throw new ArgumentException($"Length of line {i} not equal. Got {line.Length} but expected {lines[0].Length}.");
173
174 for (int j = 0; j < line.Length; j++)
175 {
176 result[i, j] = characterToTile.Invoke(line[j]);
177 }
178 }
179
180 return result;
