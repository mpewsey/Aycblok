22 public static RaycastHit Raycast(Array2D<PuzzleTile> tiles, Vector2DInt position, Vector2DInt offset, PuzzleTile blockingLayers)
23 {
24 if (offset == Vector2DInt.Zero)
25 throw new ArgumentException("Offset must be non-zero vector.");
26
27 var tile = PuzzleTile.None;
28 blockingLayers |= PuzzleTile.OutOfBounds;
29
30 while ((tile & blockingLayers) == PuzzleTile.None)
31 {
32 position += offset;
33 tile = tiles.GetOrDefault(position.X, position.Y, PuzzleTile.OutOfBounds);
34 }
35
36 return new RaycastHit(tile, position);
37 }

56 public static string GetTiledString(IList<Array2D<PuzzleTile>> tiles, int columns)
57 {
58 if (tiles.Count == 0)
59 return string.Empty;
60
61 const int spaceCount = 3;
62 columns = Math.Max(columns, 1);
63 var builder = new StringBuilder();
64 var width = 2 * tiles[0].Columns + spaceCount;
65 var rows = (int)Math.Ceiling(tiles.Count / (double)columns);
66
67 for (int row = 0; row < rows; row++)
68 {
69 // Add headers
70 for (int column = 0; column < columns; column++)
71 {
72 var index = row * columns + column;
73
74 if (index >= tiles.Count)
75 break;
76
77 var header = index == 0 ? "Start board:" : $"Move {index}:";
78 builder.Append(header);
79
80 if (column < columns - 1)
81 {
82 var length = Math.Max(width - header.Length, 0);
83 builder.Append(' ', length);
84 }
85 }
86
87 builder.Append('\n');
88
89 // Add tile strings
90 for (int i = 0; i < tiles[0].Rows; i++)
91 {
92 for (int column = 0; column < columns; column++)
93 {
94 var index = row * columns + column;
95
96 if (index >= tiles.Count)
97 break;
98
99 // Add row tile strings
100 for (int j = 0; j < tiles[0].Columns; j++)
101 {
102 builder.Append(GetTileCharacter(tiles[index][i, j])).Append(' ');
103 }
104
105 // Add separator
106 if (column < columns - 1)
107 builder.Append(' ', spaceCount);
108 }
109
110 builder.Append('\n');
111 }
112
113 if (row < rows - 1)
114 builder.Append("\n");
115 }
116
117 return builder.ToString();

143 private static char GetTileCharacter(PuzzleTile tile)
144 {
145 var layers = PuzzleTile.StopBlock | PuzzleTile.BreakBlock | PuzzleTile.PushBlock | PuzzleTile.Goal | PuzzleTile.Void;
146
147 switch (tile & layers)
148 {
149 case PuzzleTile.None:
150 return '.';
151 case PuzzleTile.StopBlock:
152 return '#';
153 case PuzzleTile.BreakBlock:
154 return '%';
155 case PuzzleTile.PushBlock:
156 return 'o';
157 case PuzzleTile.Goal:
158 return '$';
159 case PuzzleTile.Goal | PuzzleTile.PushBlock:
160 return '*';
161 case PuzzleTile.Void:
162 return '!';
163 case PuzzleTile.BlockVoid:
164 return '+';
165 case PuzzleTile.PusherVoid:
166 return '@';
167 default:
168 throw new ArgumentException($"Unhandled tile type: {tile}.");
169 }
170 }
