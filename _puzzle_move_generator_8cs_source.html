121 public PuzzleLayout GenerateLayout(Array2D<PuzzleTile> tiles, RandomSeed randomSeed)
122 {
123 Logger.Log("[Puzzle Move Generator] Generating puzzle moves...");
124 Initialize(randomSeed);
125
126 for (int i = 1; i <= MaxIterations; i++)
127 {
128 Layout = new PuzzleLayout(tiles, randomSeed.Seed);
129 AddPushBlocks();
130
131 if (AddMoves())
132 {
133 Layout.Moves.Reverse();
134 Logger.Log($"[Puzzle Move Generator] Puzzle move generation complete in {i} attempts.");
135 return Layout;
136 }
137
138 Logger.Log("[Puzzle Move Generator] Target moves not met. Restarting...");
139 }
140
141 Logger.Log("[Puzzle Move Generator] Failed to generate puzzle moves.");
142 return null;
143 }

148 private List<Vector2DInt> FindGoalPositions()
149 {
150 var positions = Layout.Tiles.FindIndexes(x => x.HasFlag(PuzzleTile.Goal));
151 var result = new List<Vector2DInt>();
152
153 foreach (var position in positions)

167 private bool AnyNeighborTileIsEmpty(int row, int column)
168 {
169 return GetTile(row, column - 1) == PuzzleTile.None
170 || GetTile(row, column + 1) == PuzzleTile.None
171 || GetTile(row - 1, column) == PuzzleTile.None
172 || GetTile(row + 1, column) == PuzzleTile.None;
173 }

179 private void AddPushBlocks()
180 {
181 var positions = FindGoalPositions();
182
183 if (positions.Count == 0)
184 throw new ArgumentException("No valid goal positions found.");
185
186 for (int i = 0; i < PushBlockCount; i++)
187 {
188 var position = positions[RandomSeed.Next(0, positions.Count)];
189 Layout.Tiles[position] |= PuzzleTile.PushBlock;
190 Layout.PushBlockPositions.Add(position);
191 }
192 }

197 private bool AddMoves()
198 {
199 foreach (var pushBlock in GetRandomPushBlocks())
200 {
201 if (!AddMove(pushBlock))
202 return false;
203 }
204
205 for (int i = PushBlockCount; i < TargetPushCount; i++)
206 {
207 if (!AddPushBlockMove())
208 return false;
209 }
210
211 return true;
212 }

217 private bool AddPushBlockMove()
218 {
219 foreach (var pushBlock in GetRandomPushBlocks())
220 {
221 if (AddMove(pushBlock))
222 return true;
223 }
224
225 return false;

277 private bool SideTilesPermitPushBlock(Vector2DInt position, Vector2DInt offset)
278 {
279 var sidePosition1 = position + new Vector2DInt(offset.Y, offset.X);
280 var sidePosition2 = position + new Vector2DInt(-offset.Y, -offset.X);
281 var sideTile1 = GetTile(sidePosition1);
282 var sideTile2 = GetTile(sidePosition2);
283
284 return (PushBlockCanOccupy(sideTile1) && CanBeStopTile(sideTile2))
285 || (PushBlockCanOccupy(sideTile2) && CanBeStopTile(sideTile1));
286 }

292 private bool GoalInSight(Vector2DInt position)
293 {
294 var layers = PuzzleTile.PushBlock | PuzzleTile.BreakBlock | PuzzleTile.StopBlock | PuzzleTile.OutOfBounds | PuzzleTile.BlockVoid | PuzzleTile.Goal;
295
296 foreach (var offset in Offsets)
297 {
298 var hit = PuzzleBoard.Raycast(Layout.Tiles, position, offset, layers);
299
300 if (hit.HitTile.HasFlag(PuzzleTile.Goal))
301 return true;
302 }
303
304 return false;
305 }

312 private bool IsOppositeLastMove(int pushBlock, Vector2DInt offset)
313 {
314 foreach (var move in Layout.Moves.Reverse<PuzzleMove>())
315 {
316 if (move.PushBlock == pushBlock)

346 private List<Vector2DInt> GetFromPositions(int pushBlock, Vector2DInt offset)
347 {
348 var result = new List<Vector2DInt>();
349 var toPosition = Layout.PushBlockPositions[pushBlock];
350 var tile = GetTile(toPosition);
351 var forwardPosition = toPosition - offset;
352 var forwardTile = GetTile(forwardPosition);
353
354 // Ensure that a stop tile can be added if the push block is not located on a goal.
355 if (!tile.HasFlag(PuzzleTile.Goal) && !CanBeStopTile(forwardTile))
356 return result;
357
358 // Ensure that the move is not opposite the last if prevent reversals is enabled.
359 if (PreventReversals && IsOppositeLastMove(pushBlock, offset))
360 return result;
361
362 var layers = PuzzleTile.PushBlock | PuzzleTile.BreakBlock | PuzzleTile.StopBlock | PuzzleTile.OutOfBounds | PuzzleTile.BlockVoid | PuzzleTile.Goal;
363 var hit = PuzzleBoard.Raycast(Layout.Tiles, toPosition, offset, layers);
364
365 for (var fromPosition = toPosition + offset; fromPosition != hit.Position; fromPosition += offset)
366 {
367 // Ensure that the block can be pushed.
368 if (!PusherCanOccupyBackTile(fromPosition, offset))
369 continue;
370 // Ensure the push block can be moved on one of its sides, and that it can be blocked on the other.
371 if (!SideTilesPermitPushBlock(fromPosition, offset))
372 continue;
373 // Prevent location from being aligned with a goal when it is not located on a goal.
374 if (!tile.HasFlag(PuzzleTile.Goal) && GoalInSight(fromPosition))
375 continue;
376
377 result.Add(fromPosition);
378 }
379
380 RandomSeed.Shuffle(result);
381 return result;
382 }

408 private PuzzleTile GetStopTile(int pushBlock, Vector2DInt offset)
409 {
410 var position = Layout.PushBlockPositions[pushBlock];
411
412 if (GetTile(position).HasFlag(PuzzleTile.Goal))
413 return PuzzleTile.Goal;
414
415 var blockPosition = position - offset;
416 var blockTile = GetTile(blockPosition);
417
418 if (blockTile.HasFlag(PuzzleTile.PushBlock))
419 return PuzzleTile.PushBlock;
420 if (blockTile.HasFlag(PuzzleTile.StopBlock))
421 return PuzzleTile.StopBlock;
422 if (IsOppositeLastMove(pushBlock, offset) || Layout.Intersects(blockPosition))
423 return PuzzleTile.BreakBlock;
424 return PuzzleTile.StopBlock;
425 }

431 private bool AddMove(int pushBlock)
432 {
433 foreach (var offset in GetRandomOffsets())
434 {
435 foreach (var fromPosition in GetFromPositions(pushBlock, offset))
436 {
437 var stopTile = GetStopTile(pushBlock, offset);
438 var toPosition = Layout.PushBlockPositions[pushBlock];
439 var move = new PuzzleMove(pushBlock, stopTile, fromPosition, toPosition);
440 move.InverseApply(Layout.Tiles);
441 Layout.PushBlockPositions[pushBlock] = fromPosition;
442 Layout.Moves.Add(move);
443 return true;
444 }
445 }
446
447 return false;
448 }
